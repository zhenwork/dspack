--fname raw_data.dsdata \
--fsave cleaned_data.dsdata \
--read_dname image_file \

--remove_bad_pixels algorithm=2 sigma=5 window_size_px=11 \
--subtract_background \
--parallax_correction \
--polarization_correction \
--solid_angle_correction \
